    map()

    An example of map 

  1. 
function App() {
const languages = ['JS', 'Python', 'Java', 'C', 'C++', 'C#'];
return (
    <div className="App">
      {languages.map((language) => {
        return <div>I love {language}</div>
      })}
    </div>
  );
}
